 151     * @param boolean arm `true` will arm, `false` will disarm the device.
 152     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 
 153     */
 154    arm(arm:boolean):Promise {
 155        const audio = this.getAudible();
 156        const cmds = [
 157        /* motionDetect*/ `cmd=setmdattr&enable=${arm?1:0}&sensitivity=${videoSensitivity}&left=0&top=0&right=1920&bottom=1080&index=0&name=MD0`,
 158        /* audioDetect */ `cmd=setaudioalarmattr&aa_enable=${arm?1:0}&aa_value=${audioSensitivity}`,
 159        /* audioAlert  */ `cmd=setalarmact&aname=alarmbeep&switch=${audio?'on':'off'}`,
 160        /* audioTime   */ `cmd=setalarmbeepattr&audiotime=5`,
 161        /* emailsnap   */ `cmd=setalarmact&aname=emailsnap&switch=off`,
 162        /* SDcardSnap  */ `cmd=setalarmact&aname=snap&switch=off`,
 163        /* SDcardRecord*/ `cmd=setalarmact&aname=record&switch=off`,
 164        /* ftpSnap     */ `cmd=setalarmact&aname=ftpsnap&switch=on`,
 165        /* ftpRecord   */ `cmd=setalarmact&aname=ftprec&switch=on`,
 166        /* preset      */ `cmd=setalarmact&aname=preset&switch=off`,
 167        /* relay       */ `cmd=setrelayattr&time=5&cmd=setalarmact&aname=relay&switch=off`,
 168        /* motor       */ `cmd=setmotorattr&alarmpresetindex=1`,
 169        /* other       */ `cmd=setalarmact&aname=type&switch=off`
 170        ];
 171        
 172        const cmd = `${this.path}alarm.cgi?${cmds.join('&')}`;
 173        return this.sendCommandToDevice(cmd)        
 174            .then((result) => this.armed = result)  // resolves to the arming status of the device (true or false)
 175            .then((result) => {
 176                const successes = result.data.split('\n');
 177                let success = successes[0] === 'Success';
 178                log.debug(`individual arm results: (${typeof result.data}) ${log.inspect(result.data)}`);
 179                successes.forEach((s:string, i:number) => { if(s && s!=='Success') { log.warn(`Command '${cmds[i]}' reported error '${s}'`); }});
 180                log.debug(`arm result: ${success? 'successful' : 'error'}`);
 181                if (success!==true) {
 182                    log.warn(`received data: ${log.inspect(result.data, null)}`);
 183                    success = result.data.indexOf('Success') === 0;
 184                    if (success!==true) {
 185                        log.error(`setting motion detect not successful`);
 186                    }
 187                } 
 188                return success;
 189            })
 190            .catch(err => {
 191                log.error(err);
 192                return this.armStatus();
 193            });
 194    }
