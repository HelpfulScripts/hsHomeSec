  13function addCommands() {
  14    Comm.addCommand(Exec.helpFn,        'help');
  15    Comm.addCommand(Exec.facetimeFn,    'facetime');
  16    Comm.addCommand(Exec.armFn,         'arm',      '[away]');
  17    Comm.addCommand(Exec.disarmFn,      'disarm');
  18    Comm.addCommand(Exec.disarmFn,      'relax');
  19    Comm.addCommand(Exec.armingStatusFn,'status');
  20    Comm.addCommand(Exec.snapFn,        'snap');
  21    Comm.addCommand(Exec.lightFn,       'light',    'on|off');
  22    Comm.addCommand(Exec.restartFn,     'restart');
  23    log.debug(`added ${Comm.getCommands().length} commands`);
  24}

  26function createDevices(settings:Settings) {
  27    settings.devices.map((dev:DeviceSettings) => {
  28        if (dev.type === 'foscam')   { new Foscam(dev, settings); }
  29        if (dev.type === 'wansview') { new WansView(dev, settings); }
  30        log.debug(`created device '${dev.name}'`);
  31    });
  32}

  37export const startSecuritySystem = (settings:Settings):Settings => {
  38    log.logFile(`${settings.homeSecDir}${settings.logDir}${settings.logFile}`);
  39    settings.users.map(user => users.addUser(user));
  40    users.setDefaultRecipient(settings.activeRecipient);
  41    createDevices(settings);
  42    addCommands();
  43    new Comm.EmailPolling(5000);
  44    log.info('security system started');
  45    return settings;
  46};
