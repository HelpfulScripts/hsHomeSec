  45     * @param string deviceName the name of the device

  49        return this.sendCommandToDevice(cmd)
  50        .then((res:http.HttpResponse) => {
  51            let src = res.body.html.body.img.attrs.src;
  52            if (src.indexOf('../')===0) { src = src.substr(2); }
  53            this.log.debug(`get snapshot: ${src}`);
  54            return this.sendCommandToDevice(src);    // no command word triggers simple request for options.path

  57    }

  72                this.log.error(`ftp config result=${result.result}: \n${this.log.inspect(result)}`);

  74            }
  75        })

  80     * promises to set the device's ftp confguration.

  86        this.log.info(`setting FTP config`);
  87        return this.sendCommandToDevice(cmd)
  88            .then((res:http.HttpResponse) => {

  90                this.log.info(`setFtpCfg ${success?'success':'failure'}`);
  91                this.log.debug(`res: ${this.log.inspect(res.body, null)}`);
  92                return success;
  93            })
  94            .catch(err => {
  95                this.log.error(err);
  96                return false;
  97            });
  98    }

 103     * @param function cb the function to call with the result

 118    /**
 119     * promises to check whether the device is armed or not. 
 120     * @param string deviceName the name of the device
 121     * @param function cb the function to call with the result
 122     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 
 123     */
 124    armStatus():Promise {
 125        const cmd = `${this.path}getDevState`;
 126        // results in {motionDetectAlarm: '0'- disarmed, '1'-no alarm, '2'- detect alarm}
 127        return this.sendCommandToDevice(cmd)  
 128            // resolves to True (armed) of False (disarmed)
 129            .then((result:any) => this.armed = (result.body.motionDetectAlarm !== '0'))   
 130            .catch(err => {
 131                this.log.error(err);
 132                throw err;
 133            });
 134    }

 136    /**
 137     * returns a promise to arm or disarm the device.
 138     * @param string deviceName the name of the device
 139     * @param boolean arm `true` will arm, `false` will disarm the device.
 140     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 
 141     */
 142    arm(arm:boolean):Promise {
 143        if (!this.getSettings().useAlarm) { 
 144            this.armed = false;
 145            return Promise.resolve(false); 
 146        }
 147        const link = linkage.pic + linkage.video + (this.getAudible()? linkage.audio : 0);
 148        const cmd = `${this.path}${armCmd}&isEnable=${arm?1:0}&linkage=${link}&sensitivity=${sensitivity}&snapInterval=${snapInterval}&triggerInterval=5`;
 149        return this.sendCommandToDevice(cmd)        
 150            .then((res) => {
 151                const success = res.body.CGI_Result.result === '0';
 152                this.log.debug(`arm result: ${success? 'successful' : 'error'}`);
 153                if (!success) { 
 154                    this.log.error(`received data: ${this.log.inspect(res.data, null)}`);
 155                } 
 156                return this.armed = success;
 157            })  // resolves to the arming status of the device (true or false)
 158            .catch(err => {
 159                this.log.error(err);
 160                return this.armStatus();
 161            });
 162    }
