 219        /**
 220     * promised to send `cmd` to the foscam camera specified in `options`
 221     * @param cmd the command string to send
 222     * @param dynRef an http options object
 223     */
 224    protected sendCommandToDevice(cmd:string, referer?:string):Promise {
 225        const settings = this.getSettings();
 226        const Url = new URL(`http://${settings.host}:${settings.port}${cmd}`);
 227        this.log.debug(`${this.getName()} requesting ${Url.href}`);
 228        return http.request(Url, new http.Digest(settings.user, settings.passwd), referer)
 229            .then((r:http.HttpResponse) => {
 230                this.log.debug(`${this.getName()} received ${r.response.headers['content-type']} for ${cmd}`);
 231                if (r.response.headers['content-type'].indexOf('text/') >= 0) {
 232                    r.body = http.xml2json(r.data);
 233                    this.log.debug(`response: ${this.log.inspect(r.body,null)}`);
 234                }
 235                return r;
 236            })
 237            .catch(this.log.error.bind(this.log));
 238    }

 241export class DeviceList {
 242    private static list = [];
 243    static addDevice(device: Device) {
 244        const settings = device.getSettings();
 245        if (!settings.name) { log.error('device name missing'); }
 246        if (!settings.host) { log.error('device host missing'); }
 247        if (!settings.port) { log.error('device port missing'); }
 248        DeviceList.list.push(device);
 249        // reference device by both short name and unique ID
 250        DeviceList.list[settings.name] = device;
 251        DeviceList.list[settings.id]   = device;
 252    }
 253
 254    static getDevice(name:string):Device {
 255        return DeviceList.list[name];
 256    }
 257
 258    /** returns a copy of the device list */
 259    static getDevices():Device[] {
 260        return this.list.slice(0);
 261    }
