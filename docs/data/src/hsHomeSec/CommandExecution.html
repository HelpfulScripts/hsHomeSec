  28function armingCall(deviceCalls: Promise[]):Promise {
  29    log.info(`waiting for ${deviceCalls.length} responses`);
  30    return Promise.race([
  31        Promise.all(deviceCalls),   // call all alarm devices 
  32        timeout(20000)              // and timeout after 20s
  33        ])
  34    .then(() => { log.info(`...completed`); })
  35    .catch(err => {
  36        log.error(err);
  37        const resp = 'some device status results missing';
  38        log.error(resp);
  39    });
  40}

  67/**
  68 * request to snap a picture. If a device name is specified, a snapshot from that device will be requested.
  69 * If no name is specified, a snapshot from each available camera will be requested.
  70 * @param params : `[]` optional list of device names to get snapshots from
  71 * @return promise to provide the file name if successful
  72 */
  73export const snapFn = (params?:string[]):Promise<{attachments:string[]}> => {
  74    const getSnap = (dev:Camera): Promise => 
  75        !dev.hasVideo()? Promise.resolve(undefined) :
  76            dev.snapPicture()
  77            .then(picData => {
  78                let fileName = path.normalize(gSnapshotDir + date(`${dev.getName()}_%YYYY%MM%DD-%hh-%mm-%ss.jpg`));
  79                log.info(`saving snapshot from ${dev.getName()} at ${fileName}`);
  80                return fs.writeStream(fileName, picData.data);
  81            });
  82        log.debug(`snap '${params?params[0]:''}': ${DeviceList.getDevices().map(d=>d.getName()).join(', ')}`);
  83        return Promise.all(                 // get snapshot from all devices    :  get snapshot from specific device
  84        (!params || !params[0] || params[0] === '')? DeviceList.getDevices().map(getSnap) : [getSnap(DeviceList.getDevice(params[0]))]
  85    ).then((files) => { 
  86        return {attachments:files}; }); // send result(s) back to user
  87};

 125/**
 126 * arms all armable devices. If `away` is specified, it also sets the device's audible alarm repsonses 
 127 * @param param `[away]` optional parameter; 'away' also sets audible alarm
 128 */
 129export const armFn = (param?:string[]):Promise<{message:string}> => {
 130    const audible = (param && param[0] === 'away');
 131    const devices = DeviceList.getDevices().filter(d => d.hasAlarm());
 132
 133    return Promise.all(
 134        devices.map((d:AlarmDevice) => 
 135            d.setAudible(audible)
 136            .then(() => d.arm(true))
 137            .then((b:boolean) => `${d.getName()} ${b?'armed':'??'} ${d.getAudible()?'with siren':''}`)
 138        )
 139    )
 140    .then((results:string[]) => {
 141        log.debug(`devices armed: ${log.inspect(results)}`);
 142        return {message: results.join('\n')};
 143    });
 144};

 162 * requests the status of attached sensors.
 163 * @param param [] optional device name list. If missing, status for all sensors will be requested
 164 * @return a literal {: }
 165 */
 166export const armingStatusFn = (param?:string[]):Promise<{message:{[x:string]:boolean}[]}> => {
 167    const alarmDevices = (param? param.map(d => DeviceList.getDevice(d)) : DeviceList.getDevices())
 168        .filter((dev:Device) => dev.hasAlarm());

 170    .then(() => { return { message: alarmDevices.map((dev:AlarmDevice) => { return {[dev.getName()]: dev.isArmed()};})};});
