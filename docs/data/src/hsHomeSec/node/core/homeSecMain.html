  29function cli(args:string[]): Promise {

  30    args.forEach((arg:string) => {

  31        const cmd = arg.split('=');

  32        if (cmd[0] === 'debug')    { log.level(log.DEBUG, true); }

  33        if (cmd[0] === 'info')     { log.level(log.INFO, true); }

  34        if (cmd[0] === 'warning')  { log.level(log.WARN, true); }

  35        if (cmd[0] === 'ftp')      { cliParams.ftpServer = true; }

  36    });

  37    return Promise.resolve();

  38}

  40function ftpInit(settings: CfgSettings):CfgSettings {

  41    if (cliParams.ftpServer) { ftp.start(settings.homeSecDir, settings.ftp); }

  42    return settings;

  43}

  50async function start() {

  51    log.debug('Starting Home Security System');

  52    log.level(log.INFO);

  53    await cli(process.argv);

  54    const cfg = await fs.readJsonFile(__dirname+'/../../config/homeCfg.json');

  55    await ftpInit(cfg);

  56    // await httpInit(cfg);

  57    await init.startSecuritySystem(cfg);

  58    await init.initDevices(cfg);

  59    await setAlarmText(cfg);

  60    await init.startScheduledTasks(cfg);

  61}
