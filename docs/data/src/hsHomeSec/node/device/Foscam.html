  28    constructor(device: DeviceSettings, settings:CfgSettings) {

  32        let schedule    = '';

  35        for (let i=0; i<10; i++) { area = area + '&area' + i + '=1023'; }

  43    async setTime() {

  44        // timeFormat: 0:YYYY-MM-DD, 1:DD/MM/YYYY, 2:MM/DD/YYYY

  45        const date = new Date();

  46        const tz = 0; // date.getTimezoneOffset()*60; set local time

  47        const yr = date.getFullYear();

  48        const m  = date.getMonth()+1;

  49        const d  = date.getDate();

  50        const h  = date.getHours();

  51        const min = date.getMinutes();

  52        const sec = date.getSeconds();

  53        const cmd = `${this.path}setSystemTime&timeSource=1&date Format=0&timeFormat=1&timeZone=${tz}&isDst=0&dst=1&year=${yr}&mon=${m}& day=${d}&hour=${h}&minute=${min}&sec=${sec}`;

  54        await this.sendCommandToDevice(cmd);

  55    }

  57    /**

  58     * captures a snapshot from the device and saves it to gSnapshotDir

  59     * @param string deviceName the name of the device

  60     */

  61    snapPicture():Promise {

  62        const cmd = `${this.path}snapPicture`;

  63        return this.sendCommandToDevice(cmd)

  64        .then((res:http.HttpResponse) => {

  65            let src = res.body.html.body.img.attrs.src;

  66            if (src.indexOf('../')===0) { src = src.substr(2); }

  67            this.log.debug(`get snapshot: ${src}`);

  68            return this.sendCommandToDevice(src);    // no command word triggers simple request for options.path

  69        })

  70        .catch(this.log.error.bind(this.log));

  71    }

  73    /**

  74     * gets the device's ftp confguration and calls `cb`. 

  75     */

  76    getFtpCfg():Promise {

  77        const cmd = `${this.path}getFtpConfig`;

  78        return this.sendCommandToDevice(cmd)

  79        .then((res:http.HttpResponse) => { 

  80            const result = res.body.CGI_Result;

  81            if (result.result === '0') {

  82                result.ftpAddr = unescape(result.ftpAddr);

  83                this.log.info(`ftp config = \n${this.log.inspect(result)}`); 

  84                return result; 

  85            } else {

  86                this.log.error(`ftp config result=${result.result}: \n${this.log.inspect(result)}`);

  87                return result;

  88            }

  89        })

  90        .catch(this.log.error.bind(this.log));

  91    }

  93    /**

  94     * promises to set the device's ftp confguration. 

  95     * The promsie resolves to true orfalse, depending on the success of the call

  96     */

  97    setFtpCfg():Promise {

  98        const ftpSettings = ftp.getSettings();

  99        const cmd = `${this.path}setFtpConfig&ftpAddr=ftp://${ftpSettings.host}/&ftpPort=${ftpSettings.port}&mode=0&userName=${ftpSettings.user}&password=${ftpSettings.pwd}`;

 100        this.log.info(`setting FTP config`);

 101        return this.sendCommandToDevice(cmd)

 102            .then((res:http.HttpResponse) => { 

 103                const success = res.body.CGI_Result.result === '0';

 104                this.log.info(`setFtpCfg ${success?'success':'failure'}`);

 105                this.log.debug(`res: ${this.log.inspect(res.body, null)}`);

 106                return success;

 107            })

 108            .catch(err => {

 109                this.log.error(err);

 110                return false;

 111            });

 112    }

 115     * gets the device's ftp confguration and calls `cb`. 

 116     * @param string deviceName the name of the device

 117     * @param function cb the function to call with the result

 123        return this.sendCommandToDevice(cmd)

 124        .then((res:http.HttpResponse) => { 

 125            const result = res.body.testResult === '0';

 126            this.log[result?'info':'error'](`ftp server test ${result?'succeeded':'failed'}`);

 127            return result;

 128        })

 129        .catch(this.log.error.bind(this.log));

 132    /**

 133     * promises to check whether the device is armed or not. 

 134     * @param string deviceName the name of the device

 135     * @param function cb the function to call with the result

 136     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 

 137     */

 138    armStatus():Promise {

 139        const cmd = `${this.path}getDevState`;

 140        // results in {motionDetectAlarm: '0'- disarmed, '1'-no alarm, '2'- detect alarm}

 141        return this.sendCommandToDevice(cmd)  

 142            // resolves to True (armed) of False (disarmed)

 143            .then((result:any) => this.armed = (result.body.motionDetectAlarm !== '0'))   

 144            .catch(err => {

 145                this.log.error(err);

 146                throw err;

 147            });

 148    }

 150    /**

 151     * returns a promise to arm or disarm the device.

 152     * @param string deviceName the name of the device

 153     * @param boolean arm `true` will arm, `false` will disarm the device.

 154     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 

 155     */

 156    arm(arm:boolean):Promise {

 157        if (!this.getSettings().useAlarm) { 

 158            this.armed = false;

 159            return Promise.resolve(false); 

 160        }

 161        const link = linkage.pic + linkage.video + (this.getAudible()? linkage.audio : 0);

 162        const cmd = `${this.path}${armCmd}&isEnable=${arm?1:0}&linkage=${link}&sensitivity=${sensitivity}&snapInterval=${snapInterval}&triggerInterval=5`;

 163        return this.sendCommandToDevice(cmd)        

 164            .then((res) => {

 165                const success = res.body.CGI_Result.result === '0';

 166                this.log.debug(`arm result: ${success? 'successful' : 'error'}`);

 167                if (!success) { 

 168                    this.log.error(`received data: ${this.log.inspect(res.data, null)}`);

 169                } 

 170                return this.armed = success;

 171            })  // resolves to the arming status of the device (true or false)

 172            .catch(err => {

 173                this.log.error(err);

 174                return this.armStatus();

 175            });

 176    }
