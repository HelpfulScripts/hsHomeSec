 224    setAudible(audible:boolean):Promise {

 234        /**

 235     * promised to send `cmd` to the foscam camera specified in `options`

 236     * @param cmd the command string to send

 237     * @param dynRef an http options object

 238     */

 239    protected sendCommandToDevice(cmd:string, referer?:string):Promise {

 240        const settings = this.getSettings();

 241        const Url = new URL(`http://${settings.host}:${settings.port}${cmd}`);

 242        this.log.debug(`${this.getName()} requesting ${Url.href}`);

 243        return http.request(Url, new http.Digest(settings.user, settings.passwd), referer)

 244            .then((r:http.HttpResponse) => {

 245                this.log.debug(`${this.getName()} received ${r.response.headers['content-type']} for ${cmd}`);

 246                if (r.response.headers['content-type'].indexOf('text/') >= 0) {

 247                    r.body = http.xml2json(r.data);

 248                    this.log.debug(`response: ${this.log.inspect(r.body,null)}`);

 249                }

 250                return r;

 251            })

 252            .catch(this.log.error.bind(this.log));

 253    }

 256export class DeviceList {

 257    private static list = [];

 258    static addDevice(device: Device) {

 259        const settings = device.getSettings();

 260        if (!settings.name) { log.error('device name missing'); }

 261        if (!settings.host) { log.error('device host missing'); }

 262        if (!settings.port) { log.error('device port missing'); }

 263        log.debug(`adding device '${device.getName()}'`);

 264        DeviceList.list.push(device);

 265        // reference device by both short name and unique ID

 266        DeviceList.list[settings.name] = device;

 267        if (!isNaN(settings.id)) { settings.id = 'dev_' + settings.id;}

 268        DeviceList.list[settings.id]   = device;

 269    }

 270

 271    static getDevice(name:string):Device {

 272        return DeviceList.list[name];

 273    }

 274

 275    /** returns a copy of the device list */

 276    static getDevices():Device[] {

 277        return this.list.slice(0);

 278    }
