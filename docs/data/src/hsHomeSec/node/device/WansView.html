  68    standardSend(cmd:string, name:string) {

  69        return this.sendCommandToDevice(cmd)

  70            .then((receivedData:http.HttpResponse) => { 

  71                this.log.debug(`standardSend '${cmd}' result:\n${log.inspect(receivedData.body, null)}`);

  72                if ((typeof receivedData.body === 'string') && (receivedData.body.match(/Success/))) {

  73                    this.log.info(`${name} success`);

  74                    return true;

  75                } else {

  76                    this.log.info(`${name} failure: ${log.inspect(receivedData.body, null)}`);

  77                    return false;

  78                }

  79            })

  80            .catch(err => {

  81                this.log.error('error '+err);

  82                return false;

  83            });

  84    }

 129    getFtpCfg():Promise {

 130        const cmd = `${this.path}ftp.cgi?cmd=getftpattr`;

 131        this.log.info(`setting FTP config`);

 132        return this.sendCommandToDevice(cmd)

 133            .then((receivedData:http.HttpResponse) => { 

 134                this.log.debug('get ftp config:' + log.inspect(receivedData.body));

 135                const result = {};

 136                receivedData.body.replace(/var /g, '').replace(/\n/g, '').split(';').map((p:any) => { p = p.split('='); if (p[1]) { result[p[0]] = p[1].replace(/\'/g, ''); }});

 137                this.log.info(`ftp config: \n${log.inspect(result, null)}`);

 138                return result;

 139            })

 140            .catch(err => {

 141                this.log.error('error'+err);

 142            });

 158     * @param string deviceName the name of the device

 159     * @param function cb the function to call with the result

 160     * @return Promise a promise that resolves to the testResult: 0 = success, -1 = failure, -3 = unknown 

 161     */

 162    testFtpServer():Promise {

 163        const cmd1 = `${this.path}ftp.cgi?cmd=testftp`;

 164        const cmd2 = `${this.path}ftp.cgi?cmd=testftpresult`;

 165        return this.sendCommandToDevice(cmd1)

 166            .then((receivedData:any) => { 

 167                this.log.debug('ftp server test1:' + receivedData.url + '\n' + log.inspect(receivedData));

 168                if (receivedData.testResult !== '0') { throw new Error('testFtpServer failed: ' + receivedData.testResult);}

 169                return receivedData;

 170            })

 171            .then(() => this.sendCommandToDevice(cmd2))

 172            .then((receivedData:any) => { 

 173                this.log.debug('ftp server test2:' + receivedData.url + '\n' + log.inspect(receivedData));

 174                if (receivedData.testResult !== '0') { throw new Error('testFtpServer failed: ' + receivedData.testResult);}

 175                return receivedData;

 176            })

 177            .catch(this.log.error.bind(this.log));

 180    /**

 181     * promises to check whether the device is armed or not. 

 182     * @param string deviceName the name of the device

 183     * @param function cb the function to call with the result

 184     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 

 185     */

 186    armStatus():Promise {

 187        const cmd = `${this.path}cmd=setalarmact&aname=alarmbeep&switch=on&cmd=setalarmbeepattr&audiotime=1`;

 188        // results in {motionDetectAlarm: '0'- disarmed, '1'-no alarm, '2'- detect alarm}

 189        return this.sendCommandToDevice(cmd)  

 190            // resolves to True (armed) of False (disarmed)

 191            .then((result:any) => this.armed = (result.motionDetectAlarm !== '0'))   

 192            .catch(err => {

 193                this.log.error(err);

 194                throw err;

 195            });

 196    }

 198    /**

 199     * returns a promise to arm or disarm the device.

 200     * @param string deviceName the name of the device

 201     * @param boolean arm `true` will arm, `false` will disarm the device.

 202     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 

 203     */

 204    arm(arm:boolean):Promise {

 205        const audio = this.getAudible();

 206        

 207        if (!this.getSettings().useAlarm) { 

 208            this.armed = false;

 209            return Promise.resolve(false); 

 210        }

 211        const cmd = `${this.path}${commands.alarm.cgi}${commands.alarm.cmds(arm, audio)}`;

 212        return this.sendCommandToDevice(cmd)        

 213            .then((result) => this.armed = result)  // resolves to the arming status of the device (true or false)

 214            .then((result) => {

 215                const successes = result.data.split('\n');

 216                let success = successes[0] === 'Success';

 217                this.log.debug(`individual arm results: (${typeof result.data}) ${log.inspect(result.data)}`);

 218                successes.forEach((s:string, i:number) => { if(s && s!=='Success') { this.log.warn(`Command '${commands.alarm.cmds[i]}' reported error '${s}'`); }});

 219                this.log.debug(`arm result: ${success? 'successful' : 'error'}`);

 220                if (success!==true) {

 221                    this.log.warn(`received data: ${log.inspect(result.data, null)}`);

 222                    success = result.data.indexOf('Success') === 0;

 223                    if (success!==true) {

 224                        this.log.error(`setting motion detect not successful`);

 225                    }

 226                } 

 227                return success;

 228            })

 229            .catch(err => {

 230                this.log.error(err);

 231                return this.armStatus();

 232            });

 233    }
