  11import { newLog }      from 'hsnode';   const log = newLog('hsMain');
  12import { fs }       from 'hsnode';
  13import * as init    from './Init';
  14import { Settings } from './Settings';
  15import * as ftp     from '../comm/ftpSrv';
  16import * as httpSrv from '../comm/httpSrv';
  17import { setAlarmText}  from './alarm';
  18
  19const cliParams = {
  20    ftpServer: false
  21};
  22// log.level(log.DEBUG);
  23
  24//==========================================================
  25// Settings

  28function cli(args:string[]): Promise {
  29    args.forEach((arg:string) => {
  30        const cmd = arg.split('=');
  31        if (cmd[0] === 'debug')    { log.level(log.DEBUG, true); }
  32        if (cmd[0] === 'info')     { log.level(log.INFO, true); }
  33        if (cmd[0] === 'warning')  { log.level(log.WARN, true); }
  34        if (cmd[0] === 'ftp')      { cliParams.ftpServer = true; }
  35    });
  36    return Promise.resolve();
  37}
  38
  39function ftpInit(settings: Settings):Settings {
  40    if (cliParams.ftpServer) { ftp.start(settings.homeSecDir, settings.ftp); }
  41    return settings;
  42}
  43
  44function httpInit(settings: Settings):Settings {
  45    httpSrv.start(); 
  46    return settings;
  47}
  48
  49try {
  50    log.debug('Starting Home Security System');
  51    log.level(log.INFO);
  52    cli(process.argv)
  53    .then(() => fs.readJsonFile(__dirname+'/../../config/homeCfg.json'))
  54    .then(ftpInit)
  55    .then(httpInit)
  56    .then(init.startSecuritySystem)
  57    .then(init.initDevices)
  58    .then(setAlarmText)
  59    .catch(log.error); 
  60
  61    process.on('exit', (code:string) => {
  62        log.info(`About to exit with code: ${code}`);
  63        httpSrv.stop();
  64    });
  65}
  66catch(err) { log.error(err); }
