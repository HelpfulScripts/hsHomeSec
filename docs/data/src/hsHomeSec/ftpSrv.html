   1import { FtpSrv }       from 'ftp-srv';
   2import { Log }          from 'hsnode';  const log = new Log('ftpSrv');
   3import { fs }           from 'hsnode'; 
   4import { raiseAlarm }   from '../core/alarm';

   6export interface FtpSettings {
   7    host: string;
   8    port: number;
   9    root: string;
  10    user: string;
  11    pwd:  string;
  12}
  13
  14const settings:FtpSettings = {
  15    host: '0.0.0.0',    // will be set during start via cfg file
  16    port:  21,          
  17    root: '',           // will be set during start()
  18    user: 'one',
  19    pwd:  'two'
  20};
  21
  22export function getSettings():FtpSettings {
  23    return settings;
  24}

  26function login(data:any, resolve:any, reject:any) {
  27    data.connection.on('RETR', (error:string, filePath:string) => { 
  28        if (error) { log.error(`reading '${filePath}': ${error}`); }
  29              else { log.warn(`reading '${filePath}'`);}
  30    }); 
  31    data.connection.on('STOR', (error:string, filePath:string) => { 
  32        raiseAlarm(filePath);
  33        if (error) { log.error(`writing '${filePath}': ${error}`); }
  34        else { 
  35        }
  36    }); 
  37    log.debug(`ftp login received: resolving for root "${settings.root}"`);
  38    if (data.username !== settings.user || data.password !== settings.pwd) {
  39        log.error(`wrong user/pwd: ${data.username}/${data.password}`);
  40        reject(new Error('nono'));
  41    }
  42    log.debug(`login accepted for root "${settings.root}"`);
  43    resolve({root:settings.root, cwd:'./'});
  44}
  45
  46export function start(baseDir:string, s:FtpSettings): Promise { 
  47    settings.host = s.host;
  48//    settings.port = Math.floor(Math.random()*2000 + 1000);
  49    const ftpServer = new FtpSrv(`ftp://${settings.host}:${settings.port}`);
  50    const root = `${baseDir}/${s.root}`;
  51    log.debug(`testing ftp server root '${root}'`);
  52    return fs.realPath(root)
  53    .then((p:string):void => {
  54        settings.root = p;
  55        log.info(`ftp root ${settings.root}`);    
  56        ftpServer.on('login', login);  
  57        ftpServer.on('client-error', (data:any) => { 
  58            log.error(`client error received: context ${data.context} \n${log.inspect(data.error)}`);
  59        });
  60        return ftpServer.listen();
  61    })
  62    .then(() => { log.info(`ftp server started on ${settings.host}:${settings.port}`); })
  63    .catch((err:any) => {
  64        log.error(`creating ftp server on '${settings.host}:${settings.port}:/${settings.root}'`);
  65        log.error(err);
  66        ftpServer.close();
  67    });
  68}
