  27function createDevices(settings:CfgSettings) {
  28    settings.devices.map((dev:DeviceSettings) => {
  29        if (dev.type === 'foscam')   { new Foscam(dev, settings); }
  30        if (dev.type === 'wansview') { new WansView(dev, settings); }
  31        log.debug(`created device '${dev.name}'`);
  32    });
  33}

  36function wifiCheck(settings:CfgSettings) {
  37    const ssid = settings.wifiNetwork;
  38    const pwd  = settings.wifiPasswd;
  39    return async () => {
  40        const result = await node.exec('networksetup -getairportnetwork en0');
  41        log.debug(`wifi stdout:'${result.stdout}'`);
  42        log.debug(`wifi stderr:'${result.stderr}'`);
  43        if (result.stdout.match(ssid)) {
  44            log.info(`wifi network connected to ${ssid}`);
  45        } else {
  46            log.warn(`wifi not connected to ${ssid}; attempting reconnect`);
  47            log.warn(`wifi stdout:'${result.stdout}'`);
  48            log.warn(`wifi stderr:'${result.stderr}'`);
  49            const nwResult = await node.exec(`networksetup -setairportnetwork en0 ${ssid} ${pwd}`);
  50            log.warn(`reconnect stdout:'${nwResult.stdout}'`);
  51            log.warn(`reconnect stderr:'${nwResult.stderr}'`);
  52        }
  53    };
  54}

  63export const startSecuritySystem = (settings:CfgSettings):CfgSettings => {
  64    log.logFile(`${settings.homeSecDir}${settings.logDir}${settings.logFile}`);
  65    settings.users.map(user => users.addUser(user));
  66    users.setDefaultRecipient(settings.activeRecipient);
  67    createDevices(settings);
  68    addCommands();
  69    new Comm.EmailPolling(5000);
  70    log.info('security system started');
  71    return settings;
  72};

  79export const initDevices = (settings:CfgSettings):CfgSettings => {
  80    Exec.setSnapshotDir(`${settings.homeSecDir}/${settings.recDir}/`);
  81    DeviceList.getDevices().map((dev:Device) => dev.initDevice(settings));
  82    log.info(`devices initialized: ${DeviceList.getDevices().map(d=>d.getName()).join(', ')}`);
  83    return settings;
  84};

  86export function startScheduledTasks(settings:CfgSettings) {
  87    const WifiCheckHours = 1;
  88    const SetTimeHours = 12;
  89    if (settings.wifiNetwork && settings.wifiPasswd) {
  90        setInterval(wifiCheck(settings), WifiCheckHours*60*60*1000);
  91    }
  92    setInterval(setDevicesTime, SetTimeHours*60*60*1000);
  93    log.info('scheduled tasks started.');
